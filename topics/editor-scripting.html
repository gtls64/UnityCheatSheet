Editor Scripting

Create a custom editor window using the EditorWindow class.

using UnityEditor;
using UnityEngine;

public class MyEditorWindow : EditorWindow
{
    [MenuItem("Window/My Editor Window")]
    public static void ShowWindow()
    {
        GetWindow<MyEditorWindow>("My Editor Window");
    }

    void OnGUI()
    {
        GUILayout.Label("Hello, World!", EditorStyles.boldLabel);
        if (GUILayout.Button("Press Me"))
        {
            Debug.Log("Button Pressed");
        }
    }
}

Create a custom inspector using the Editor class.

using UnityEditor;
using UnityEngine;

[CustomEditor(typeof(MyScript))]
public class MyScriptEditor : Editor
{
    public override void OnInspectorGUI()
    {
        DrawDefaultInspector();

        MyScript myScript = (MyScript)target;
        if (GUILayout.Button("My Custom Button"))
        {
            // Button action
        }
    }
}
