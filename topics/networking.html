Networking

Set up basic networking using the NetworkManager class.

using UnityEngine.Networking;

public class MyNetworkManager : NetworkManager
{
    public override void OnServerConnect(NetworkConnection conn)
    {
        // Called when a client connects to the server
    }

    public override void OnClientConnect(NetworkConnection conn)
    {
        // Called when this client connects to a server
    }
}

Send messages over the network using the MessageBase class.

public class MyMessage : MessageBase
{
    public int value;
}

public class MyNetworkManager : NetworkManager
{
    public void SendMessageToServer()
    {
        MyMessage msg = new MyMessage { value = 42 };
        NetworkServer.SendToAll(MyMessage.msgType, msg);
    }
}
