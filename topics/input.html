Input

Detect key presses using the Input.GetKeyDown method.

if (Input.GetKeyDown(KeyCode.Space))
{
    // Space key was pressed
}

Detect mouse button presses using the Input.GetMouseButtonDown method.

if (Input.GetMouseButtonDown(0))
{
    // Left mouse button was pressed
}

Detect touch input using the Input.touchCount and Input.GetTouch methods.

if (Input.touchCount > 0)
{
    Touch touch = Input.GetTouch(0);
    // Process touch
}
